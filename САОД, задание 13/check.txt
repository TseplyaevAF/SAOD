// Функция check проверяет начальное условие и возвращает true,
// если (a | b) mod (b & c) != 0, и тогда функция F
// должна принять в параметры действительные значения чисел a,b,c,
// иначе их целые части
bool check(float a, float b, float c) {
int a1 = int(a);
	int b1 = int(b);
	int c1 = int(c);
	if (((a1 | b1) ^ (b1 & c1)) != 0) return 1;
	else 0;
}
